df['forecast'] = np.nan
for i in range(36, df.shape[0]):
    df.at[i, 'forecast'] = df.Temperature[range(i%36, i, 36)].mean()
    
plt.plot(df.Temperature)
plt.plot(df.forecast, color='red')
plt.legend(["time since", "forecast moving average"])
plt.show()
